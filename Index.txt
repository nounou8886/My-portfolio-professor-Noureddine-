  };

  function switchLang(lang){
    document.getElementById('greet').textContent = content[lang].greet;
    document.getElementById('heroDesc').textContent = content[lang].heroDesc;
    document.getElementById('eduTitle').textContent = content[lang].eduTitle;
    document.getElementById('skillsTitle').textContent = content[lang].skillsTitle;
    document.getElementById('servicesTitle').textContent = content[lang].servicesTitle;

    // direction for arabic
    if(lang === 'ar'){
      document.documentElement.lang = 'ar';
      document.body.classList.add('rtl');
    } else {
      document.documentElement.lang = lang;
      document.body.classList.remove('rtl');
    }

    // button styles
    document.querySelectorAll('.lang-toggle button').forEach(b=>{
      b.classList.toggle('primary', b.dataset.lang === lang);
    });
  }

  // wire language buttons
  document.getElementById('enBtn').addEventListener('click', ()=>switchLang('en'));
  document.getElementById('frBtn').addEventListener('click', ()=>switchLang('fr'));
  document.getElementById('arBtn').addEventListener('click', ()=>switchLang('ar'));
  // initialize
  switchLang('en');

  // download CV: create a simple text CV and download
  document.getElementById('downloadCV').addEventListener('click', ()=>{
    const cvText = [
      "Noureddine Lazher Mouelhi",
      "Email: nounou8886@gmail.com",

      "Profile:",
      "Professor of Translation and Computer Science. Research interests: machine translation evaluation, computational linguistics, translation technologies.",
      "",
      "Skills: Python (NLP), JavaScript, HTML/CSS, CAT tools (Trados, MemoQ), MT evaluation, curriculum design.",
      "",
      "Selected Projects:",
      "- Collaborative Glossary Web App",
      "- MT Evaluation Suite",
      "- Interactive Course Website",
      "",
      "Available for workshops, research collaborations, consultancy.",
      "",
      "Generated: " + new Date().toLocaleString()
    ].join("\\n");

    const blob = new Blob([cvText], {type: 'text/plain;charset=utf-8'});
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url;
    a.download = 'CV_Noureddine_Lazher_Mouelhi.txt';
    document.body.appendChild(a);
    a.click();
    a.remove();
    URL.revokeObjectURL(url);
